[docs]def apidoc_new_py(code:str,output:str,source_dir:str,*,project_name:str,author:str, version:str) -> None:
    """为python项目构造api文档.

    Args:
        code (str): 项目源码位置
        output (str): html文档位置
        source_dir (str): 文档源码位置
        project_name (str): 项目名
        author (str): 项目作者
        version (str): 项目版本

    """
    def new_succ_cb(content:str)->None:
        default_succ_cb(content)
        print(f"在{source_dir}创建api文档源文件成功")
        with open(Path(source_dir).joinpath("conf.py"),"r",encoding="utf-8") as fr:
            content = fr.read()
        with open(Path(source_dir).joinpath("conf.py"),"w") as fw:
            new_content= content+"""
extensions += ['recommonmark', 'sphinx.ext.napoleon', 'sphinx.ext.mathjax']
locale_dirs = ['locale/']   # path is example but recommended.
gettext_compact = False     # optional.
"""
            fw.write(new_content)
        print('完成初始化文档源文件')
        print("编译项目文档")

        def build_succ_cb(content:str)->None:
            default_succ_cb(content)

        command = f"sphinx-build -b html {source_dir} {output}"
        run_command(command,succ_cb=build_succ_cb)


    command = f"sphinx-apidoc -F -H {project_name} -E -A {author} -V {version} -a -o {source_dir} {code}"
    run_command(command,succ_cb=new_succ_cb)
