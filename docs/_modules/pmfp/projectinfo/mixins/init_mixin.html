[docs]    def init_project(self):
        """创建项目."""
        try:
            self._init_template()
            if self.form.compiler in ("python", 'node'):
                self._init_env()
            self._init_readme()

        except Exception as e:
            traceback.print_exc()
            print(str(e))
            self.clean(total=True)
            raise e
        else:

            return True
